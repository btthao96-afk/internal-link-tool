  Input: Website URL

    ↓
  Store in Database: pages_table
    ↓

┌─────────────────────────────────────────────────────────┐
│ PHASE 2: CONTENT ANALYSIS                               │
└─────────────────────────────────────────────────────────┘
  For each page:
    ↓
  [NLP Processing]:
    • Tokenize content
    • Extract keywords (TF-IDF)
    • Identify key topics
    • Generate embeddings (optional)
    ↓
  Store: keywords_table, topics_table, embeddings_table
    ↓

┌─────────────────────────────────────────────────────────┐
│ PHASE 3: RELATIONSHIP ANALYSIS                          │
└─────────────────────────────────────────────────────────┘
  For each page:
    ↓
  Find related pages:
    • Keyword overlap score
    • Semantic similarity
    • Topic matching
    ↓
  Rank by relevance (0-100)
    ↓
  Filter: score > threshold (e.g., 70%)
    ↓
  Store: relationships_table
    ↓

┌─────────────────────────────────────────────────────────┐
│ PHASE 4: INTERNAL LINK SUGGESTIONS                      │
└─────────────────────────────────────────────────────────┘
  For each page content:
    ↓
  Identify link placement opportunities:
    • Mid-content paragraphs (best for SEO)
    • Related sections
    • Avoid: first paragraph, footer
    ↓
  For each placement opportunity + related page:
    ↓
  Generate natural anchor text:
    • Use target page title/H1
    • Synonyms
    • Relevant phrases from content
    ↓
  Calculate insertion point context
    ↓
  Create suggestion object:
    {
      from_page_url,
      to_page_url,
      anchor_text,
      suggested_paragraph,
      relevance_score,
      link_type (related/topical/pillar)
    }
    ↓
  Store: suggestions_table
    ↓

┌─────────────────────────────────────────────────────────┐
│ PHASE 5: DASHBOARD & APPROVAL                           │
└─────────────────────────────────────────────────────────┘
  Display suggestions:
    • Group by source page
    • Show relevance score
    • Preview: "Page title [anchor text] → Target page"
    ↓
  User actions:
    • ✅ Approve → Generate HTML/API call
    • ❌ Reject → Store feedback
    • ✏️ Edit → Modify anchor text
    ↓

    • Option 1: API call to inject into CMS

CREATE TABLE pages (
  id SERIAL PRIMARY KEY,
  url VARCHAR(500) UNIQUE,
  title VARCHAR(200),
  h1 TEXT,

  content TEXT,
  crawled_at TIMESTAMP,

CREATE TABLE keywords (
  id SERIAL PRIMARY KEY,
  page_id INT REFERENCES pages(id),
  keyword VARCHAR(100),

);

-- Relationships table
CREATE TABLE relationships (
  id SERIAL PRIMARY KEY,
  from_page_id INT REFERENCES pages(id),
  to_page_id INT REFERENCES pages(id),
  relevance_score FLOAT,
  reason VARCHAR(100) -- "keyword_overlap", "semantic", etc

CREATE TABLE suggestions (

#### **Step 3: Crawler Module**
```javascript
// crawler.js
const axios = require('axios');
const cheerio = require('cheerio');
const { getVisitedUrls, savePageData } = require('./db');

async function crawlSite(startUrl, maxPages = 1000) {
  const queue = [startUrl];
  const visited = new Set();
  let count = 0;

  while (queue.length > 0 && count < maxPages) {
    const url = queue.shift();
    if (visited.has(url)) continue;

    try {
      const response = await axios.get(url, { timeout: 10000 });
      const $ = cheerio.load(response.data);

      // Extract data
      const data = {
        url,
        title: $('title').text(),
        h1: $('h1').text(),
        meta_desc: $('meta[name="description"]').attr('content'),
        content: $('body').text().substring(0, 5000)
      };

      await savePageData(data);

      // Find new URLs
      $('a[href]').each((_, el) => {
        const href = $(el).attr('href');
        const absoluteUrl = new URL(href, url).href;
        if (absoluteUrl.includes(new URL(startUrl).hostname)) {
          queue.push(absoluteUrl);
        }
      });

      visited.add(url);
      count++;
      console.log(`Crawled: ${url} (${count}/${maxPages})`);
    } catch (error) {
      console.error(`Error crawling ${url}:`, error.message);
    }
  }

  return count;
}

module.exports = { crawlSite };
```

#### **Step 4: NLP & Relevance Engine**
```javascript
// nlp.js
const natural = require('natural');

function extractKeywords(text, topN = 10) {
  const tokenizer = new natural.WordTokenizer();
  const tokens = tokenizer.tokenize(text.toLowerCase());

  // Remove stopwords
  const stopwords = ['the', 'is', 'at', 'which', 'on', 'and', 'or'];
  const filtered = tokens.filter(t => !stopwords.includes(t));

  // Count frequency
  const freq = {};
  filtered.forEach(token => {
    freq[token] = (freq[token] || 0) + 1;
  });

  // Sort by frequency
  return Object.entries(freq)
    .sort((a, b) => b[1] - a[1])
    .slice(0, topN)
    .map(entry => entry[0]);
}

function calculateRelevance(keywords1, keywords2) {
  const set1 = new Set(keywords1);
  const intersection = keywords2.filter(k => set1.has(k));
  const similarity = intersection.length / Math.max(keywords1.length, keywords2.length);

  return Math.round(similarity * 100);
}

module.exports = { extractKeywords, calculateRelevance };
```

#### **Step 5: Suggestion Generator**
```javascript
// suggestionEngine.js
async function generateSuggestions() {
  const pages = await getAllPages();

  for (const page of pages) {
    const keywords = await getKeywords(page.id);
    const relatedPages = await findRelatedPages(page.id, keywords);

    for (const related of relatedPages) {
      if (related.relevance_score < 70) continue;

      const suggestions = generateAnchorTexts(related, keywords);

      for (const anchor of suggestions) {
        await saveSuggestion({
          from_page_id: page.id,
          to_page_id: related.id,
          anchor_text: anchor,
          relevance_score: related.relevance_score
        });
      }
    }
  }
}

function generateAnchorTexts(targetPage, keywords) {
  const anchors = [
    targetPage.title,
    targetPage.h1,
    `${keywords[0]} ${targetPage.title}`,
    `related: ${targetPage.title}`,
    `learn more about ${keywords[0]}`
  ];

  return anchors.filter(a => a && a.length < 100);
}

module.exports = { generateSuggestions };
```

#### **Step 6: API Routes**
```javascript
// routes/suggestions.js
const express = require('express');
const router = express.Router();

router.get('/pending', async (req, res) => {
  const suggestions = await db.query(
    'SELECT s.*, p1.url as from_url, p2.url as to_url ' +
    'FROM suggestions s ' +
    'JOIN pages p1 ON s.from_page_id = p1.id ' +
    'JOIN pages p2 ON s.to_page_id = p2.id ' +
    'WHERE s.status = "pending" ' +
    'ORDER BY s.relevance_score DESC'
  );
  res.json(suggestions);
});

router.post('/approve/:id', async (req, res) => {
  await db.query('UPDATE suggestions SET status = ? WHERE id = ?',
    ['approved', req.params.id]);
  res.json({ success: true });
});

router.post('/reject/:id', async (req, res) => {
  await db.query('UPDATE suggestions SET status = ? WHERE id = ?',
    ['rejected', req.params.id]);
  res.json({ success: true });
});

module.exports = router;
```

## 7. Best Practices SEO

✅ **Do:**
- Link từ high-traffic pages → supporting pages
- Use natural, descriptive anchor text
- Link to relevant content (semantic similarity)
- Maintain link distribution (không quá nhiều link)
- Update existing internal links

❌ **Don't:**
- Over-optimize anchor text (keyword stuffing)
- Create random links không liên quan
- Link từ footer/sidebar (low value)
- Sử dụng generic anchor text ("click here")
